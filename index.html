    音を鳴らす時間（分:秒）:

    <input type="number" id="swAlarmMin" value="0" min="0"> 分

    <input type="number" id="swAlarmSec" value="5" min="0"> 秒

  </div>


  <!-- タイマー -->

  <div class="section">

    <h2>タイマー</h2>

    時間設定:

    <input type="number" id="timerMin" value="1" min="0"> 分

    <input type="number" id="timerSec" value="0" min="0"> 秒

    <div class="display" id="timerDisplay">01:00</div>

    <button onclick="startTimer()">開始</button>

    <button onclick="stopTimer()">停止</button>

    <button onclick="resetTimer()">リセット</button>

    <br>

    自動繰り返し:

    <input type="checkbox" id="autoRepeat">

    <label for="autoRepeat">ON</label>

  </div>


  <!-- 音 -->

  <audio id="bell" src="https://www.soundjay.com/misc/sounds/bell-ringing-05.mp3"></audio>


  <script>

    // ストップウォッチ

    let swTime = 0;

    let swInterval;

    function updateStopwatch() {

      const min = String(Math.floor(swTime / 60)).padStart(2, '0');

      const sec = String(swTime % 60).padStart(2, '0');

      document.getElementById('stopwatchDisplay').textContent = `${min}:${sec}`;


      const alarmMin = parseInt(document.getElementById('swAlarmMin').value);

      const alarmSec = parseInt(document.getElementById('swAlarmSec').value);

      const alarmTotal = alarmMin * 60 + alarmSec;

      if (swTime === alarmTotal) {

        document.getElementById('bell').play();

      }


      swTime++;

    }

    function startStopwatch() {

      clearInterval(swInterval);

      swInterval = setInterval(updateStopwatch, 1000);

    }

    function stopStopwatch() {

      clearInterval(swInterval);

    }

    function resetStopwatch() {

      clearInterval(swInterval);

      swTime = 0;

      document.getElementById('stopwatchDisplay').textContent = '00:00';

    }


    // タイマー

    let timerTime = 0;

    let timerInterval;

    function updateTimer() {

      if (timerTime <= 0) {

        document.getElementById('timerDisplay').textContent = '00:00';

        document.getElementById('bell').play();

        clearInterval(timerInterval);

        if (document.getElementById('autoRepeat').checked) {

          startTimer(); // 自動で再スタート

        }

        return;

      }

      timerTime--;

      const min = String(Math.floor(timerTime / 60)).padStart(2, '0');

      const sec = String(timerTime % 60).padStart(2, '0');

      document.getElementById('timerDisplay').textContent = `${min}:${sec}`;

    }

    function startTimer() {

      clearInterval(timerInterval);

      const min = parseInt(document.getElementById('timerMin').value);

      const sec = parseInt(document.getElementById('timerSec').value);

      timerTime = min * 60 + sec;

      updateTimer(); // 即座に表示更新

      timerInterval = setInterval(updateTimer, 1000);

    }

    function stopTimer() {

      clearInterval(timerInterval);

    }

    function resetTimer() {

      clearInterval(timerInterval);

      const min = parseInt(document.getElementById('timerMin').value);

      const sec = parseInt(document.getElementById('timerSec').value);

      const total = min * 60 + sec;

      const m = String(min).padStart(2, '0');

      const s = String(sec).padStart(2, '0');

      document.getElementById('timerDisplay').textContent = `${m}:${s}`;

      timerTime = total;

    }


    // 初期化

    resetTimer();

  </script>
